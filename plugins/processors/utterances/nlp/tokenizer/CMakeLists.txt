#                        Configure plugin_info.h.in                                  #

#------------------------------------------------------------------------------------#
#                               Include tokenizer                                    #
#------------------------------------------------------------------------------------#

include(tokenizer)


#------------------------------------------------------------------------------------#
#                               Source files                                         #
#------------------------------------------------------------------------------------#

# Lists of all the source files
include(sources)


#------------------------------------------------------------------------------------#
#                        Plug-in shared object                                       #
#------------------------------------------------------------------------------------#

speect_plugin_create()
