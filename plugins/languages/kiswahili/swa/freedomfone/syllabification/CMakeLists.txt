#                                Project name                                        #
#------------------------------------------------------------------------------------#

speect_plugin_definition(Syll_Swa_Freedomfone "SSyllabSwaFreedomfone" 1 0 0)


#------------------------------------------------------------------------------------#
#                        Configure plugin_info.h.in                                  #
#------------------------------------------------------------------------------------#

set(description "A Kiswahili syllabification class implementation for the Freedomfone project")

# Minimum required Speect Engine version 
set(major_min 1)
set(minor_min 0)

speect_plugin_configure_info(${description} ${major_min} ${minor_min})


#------------------------------------------------------------------------------------#
#                            Include lexical plug-ins                                #
#------------------------------------------------------------------------------------#

include(syllabification)
include(phoneset) 


#------------------------------------------------------------------------------------#
#                               Source files                                         #
#------------------------------------------------------------------------------------#

# Lists of all the source files
include(sources)


#------------------------------------------------------------------------------------#
#                        Plug-in shared object                                       #
#------------------------------------------------------------------------------------#

speect_plugin_create()
